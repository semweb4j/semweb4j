7   package org.ontoware.rdf2go.osgi;
8   
9   import java.util.logging.Logger;
10  
11  import org.ontoware.rdf2go.ModelFactory;
12  import org.ontoware.rdf2go.RDF2Go;
13  import org.osgi.framework.BundleActivator;
14  import org.osgi.framework.BundleContext;
15  import org.osgi.framework.ServiceEvent;
16  import org.osgi.framework.ServiceListener;
17  import org.osgi.framework.ServiceReference;
18  
19  /**
20   * Activate the RDF2Go framework.
21   * Listens to registered OSGI services and searches for registered
22   * {@link ModelFactory} instances.
23   * The found factories are compared to a configuration value, if the
24   * configured factory is found, it is set as default. This is the configuration
25   * value to set:
26   * <code>org.ontoware.rdf2go.defaultmodelfactory=(classname of modelfactory)</code>
27   *  
28   * @author sauermann
29   */
30  public class RDF2GoActivator implements BundleActivator, ServiceListener {
31      
32      /**
33       * set this variable in the OSGI config to define the default driver
34       * for RDF2Go.
35       */
36      public static final String DEFAULTMODELFACTORY_CFG = "org.ontoware.rdf2go.defaultmodelfactory";
37      
38      private BundleContext bc;
39      
40      private String defaultFactoryClassName = null;
41      
42      private static final Logger log = Logger.getLogger(RDF2GoActivator.class.getName());
43  
44      /**
45       * @see org.osgi.framework.BundleActivator#start(org.osgi.framework.BundleContext)
46       */
47      public void start(BundleContext context) throws Exception {
48          bc = context;
49          
50          defaultFactoryClassName = context.getProperty(DEFAULTMODELFACTORY_CFG);
51          if (defaultFactoryClassName == null)
52          {
53              log.warning("RDF2Go cannot find configuration value for default RDF2Go factory. " +
54                      "No Default ModelFactory will be available. Please set " +
55                  DEFAULTMODELFACTORY_CFG);
56              return;
57          }
58          
59          String filter = "(objectclass=" + ModelFactory.class.getName() + ")";
60          context.addServiceListener(this, filter);
61  
62          ServiceReference references[] = context.getServiceReferences(null, filter);
63  
64          for (int i = 0; references != null && i < references.length; i++) {
65              this.serviceChanged(new ServiceEvent(ServiceEvent.REGISTERED, references[i]));
66          }
67      }
68  
69      /**
70       * @see org.osgi.framework.BundleActivator#stop(org.osgi.framework.BundleContext)
71       */
72      public void stop(BundleContext context) throws Exception {
73          bc = null;
74      }
75  
76      /**
77       * Listen to changes in the ModelFactories.
78       * @see org.osgi.framework.ServiceListener#serviceChanged(org.osgi.framework.ServiceEvent)
79       */
80      public void serviceChanged(ServiceEvent event) {
81          if (defaultFactoryClassName == null)
82              return;
83          
84          ModelFactory factory;
85          switch (event.getType()) {
86          case ServiceEvent.REGISTERED:
87              factory = (ModelFactory) bc.getService(event
88                      .getServiceReference());
89              if (factory.getClass().getName().equals(defaultFactoryClassName))
90              {
91                  log.info("RDF2Go uses "+defaultFactoryClassName+" as default ModelFactory");
92                  RDF2Go.register(factory);
93              }
94              else
95                  bc.ungetService(event.getServiceReference());
96              break;
97          case ServiceEvent.UNREGISTERING:
98              factory = (ModelFactory) bc.getService(event
99                  .getServiceReference());
100             if (factory.getClass().getName().equals(defaultFactoryClassName))
101             {
102                 log.fine("RDF2Go unregistered the ModelFactory "+defaultFactoryClassName+" as default ModelFactory. No ModelFactory available now. The Bundle of the ModelFactory was unregistered.");
103                 RDF2Go.register((ModelFactory)null);
104             }
105             bc.ungetService(event
106                 .getServiceReference());
107             break;
108         }
109         // We do not care for modified
110         //case ServiceEvent.MODIFIED:
111     }
