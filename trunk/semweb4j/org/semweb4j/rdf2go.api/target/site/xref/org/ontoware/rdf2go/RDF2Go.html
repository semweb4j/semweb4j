21  /**
22   * Convenience class that allows registering and finding ModelFactories If you
23   * use one adaptor in your application, you should always use the default
24   * instance. If you use multiple frameworks, then you should instantiate the
25   * ModelFactory implementations directly.
26   * 
27   * <h2>Usage</h2>
28   * <p>
29   * This is the main entry point to use RDF2Go, and in particular to create
30   * Models. You have to register an adaptor once to use this. Example:
31   * </p>
32   * <code>
33   * // register adapter of implementation XXX, could be Jena or Sesame or anything.
34   * RDF2Go.register( new org.ontoware.rdf2go.impl.XXX.ModelFactoryImpl() );
35   * 
36   * // start using RDF2Go
37   * Model m = RDF2Go.getModelFactory().createModel();
38   * m.addTriple(...)
39   * </code>
40   * 
41   * For each RDF2Go adapter, an ModelFactoy implementation is needed.
42   * 
43   * User can call the {@link #register()} method to do that.
44   * 
45   * As a fall-back, when a user calls getModelFactory() but no adapter has been
46   * registered yet, RDF2Go looks for a class named
47   * org.ontoware.rdf2go.impl.StaticBinding and tries to call a method named
48   * "getModelFactory" to get a model factory. This approach has been inspired by
49   * SL4Js static bindign approach. When multiple adapters are found, the first in
50   * the class path is used. This fall-back allows one to simply pack together a
51   * bunch of JARs an be ready to used RDF2Go without any configuration and
52   * without any OSGi.
53   * 
54   * The expected class looks like this: <code><pre>
55   * package org.ontoware.rdf2go.impl;
56   * 
57   * import org.ontoware.rdf2go.ModelFactory;
58   * 
59   * public class StaticBinding {
60   *  public static ModelFactory getModelFactory() {
61   *    return new org.ontoware.rdf2go.impl.XXXX.ModelFactoryImpl();
62   *  }
63   * }
64   * </pre></code>
65   * 
66   * @author sauermann
67   * @author voelkel
68   */
69  public class RDF2Go {

71  	private static final Log log = LogFactory.getLog( RDF2Go.class );

73  	/**
74  	 * the default factory for RDF2Go. This is a singleton. The field is
75  	 * protected by design, JUnit tests inside RDF2Go may want to reset it.
76  	 */
77  	protected static ModelFactory modelFactory;
78  
79  	private static Exception instanceRegistered;
80  
81  	/**
82  	 * get the currently registered RDF2Go factory. Usually, you will use one
83  	 * adaptor in your application, register this adaptor using the .register()
84  	 * method and it is available here. If none is registered, this throws a
85  	 * RuntimeException.
86  	 * 
87  	 * @return the default RDF2Go instance.
88  	 */
89  	public static final ModelFactory getModelFactory() {
90  		checkModelFactory();
91  		return modelFactory;
92  	}
93  
94  	/**
95  	 * register an implementation of the RDF2Go framework. you can only register
96  	 * one framework inside one Java application, it will throw an Exception if
97  	 * you register more than one.
98  	 * 
99  	 * @param modelFactory
100 	 *            the factory to register. You can pass <code>null</code> to
101 	 *            unregister a modelFactory. Unregistering is restricted to
102 	 *            frameworks, you will not need it in simple applications.
103 	 * @throws RuntimeException
104 	 *             if registered already. See the cause of this exception for a
105 	 *             stacktrace where you first registered.
106 	 */
107 	public static final void register(ModelFactory modelFactory) {
108 		if (modelFactory == null) {
109 			RDF2Go.modelFactory = null;
110 			return;
111 		}
112 		if ((RDF2Go.modelFactory != null)
113 				&& (!RDF2Go.modelFactory.getClass().equals(
114 						modelFactory.getClass())))
115 			throw new RuntimeException("already registered framework "
116 					+ modelFactory.getClass().getName()
117 					+ " in the attached Exception (see stacktrace).",
118 					instanceRegistered);
119 		RDF2Go.modelFactory = modelFactory;
120 		// store the stacktrace to help people find where the falsly registered
121 		// first
122 		instanceRegistered = new Exception("registered framework before");
123 	}
124 
125 	public static final void register(String modelFactoryClassname)
126 			throws ModelException {
127 		Class c;
128 		try {
129 			c = Class.forName(modelFactoryClassname);
130 			Constructor constructor;
131 			try {
132 				constructor = c.getConstructor(new Class[] {});
133 				ModelFactory modelBuilder;
134 				try {
135 					modelBuilder = (ModelFactory) constructor.newInstance();
136 					register(modelBuilder);
137 				} catch (IllegalArgumentException e) {
138 					throw new ModelException(e);
139 				} catch (InstantiationException e) {
140 					throw new ModelException(e);
141 				} catch (IllegalAccessException e) {
142 					throw new ModelException(e);
143 				} catch (InvocationTargetException e) {
144 					throw new ModelException(e);
145 				}
146 			} catch (SecurityException e) {
147 				throw new ModelException(e);
148 			} catch (NoSuchMethodException e) {
149 				throw new ModelException(e);
150 			}
151 		} catch (ClassNotFoundException e) {
152 			throw new ModelException(e);
153 		}
154 
155 	}

157 	private static void checkModelFactory() {
158 		// before failing, try to find an adapter via a static binding
159 		// approach, like the one used by SL4J.org
160 		if (modelFactory == null) {
161 			try {
162 				Class clazz = Class
163 						.forName("org.ontoware.rdf2go.impl.StaticBinding");
164 				Method method = clazz.getMethod("getModelFactory",
165 						new Class[] {});
166 				Object result = method.invoke(clazz, new Object[] {});
167 				if (result instanceof ModelFactory) {
168 					modelFactory = (ModelFactory) result;
169 					log.info("Using ModelFactory '"
170 									+ result.getClass()
171 									+ "' which was loaded via org.ontoware.rdf2go.impl.StaticBinding.");
172 				}
173 			} catch (ClassNotFoundException e) {
174 				// TODO Auto-generated catch block
175 				e.printStackTrace();
176 			} catch (SecurityException e) {
177 				// TODO Auto-generated catch block
178 				e.printStackTrace();
179 			} catch (NoSuchMethodException e) {
180 				// TODO Auto-generated catch block
181 				e.printStackTrace();
182 			} catch (IllegalArgumentException e) {
183 				// TODO Auto-generated catch block
184 				e.printStackTrace();
185 			} catch (IllegalAccessException e) {
186 				// TODO Auto-generated catch block
187 				e.printStackTrace();
188 			} catch (InvocationTargetException e) {
189 				// TODO Auto-generated catch block
190 				e.printStackTrace();
191 			}
192 
193 		}
194 
195 		// if modelfactory is still null, give up
196 		if (modelFactory == null) {
197 			throw new IllegalStateException(
198 					"No ModelFactoy was registered. Please register one via RDF2Go.register(ModelFactory mf)");
199 		}
200 	}
