27  	/**
28  	 * This constructor should not be called by API users. You should use
29  	 * ModelImplXXX.createURI(..) instead.

32  	 * @param uriString

34  	public static URI create(String uriString) {
35  		// the following block is an internal RDF2Go debug setting.
36  		// RDF2Go users may not rely on this validation
37  		if (log.isDebugEnabled()) {
38  			// check URI: applying strict and not 100% correct URI validation
39  			try {
40  				new java.net.URI(uriString);
41  			} catch (URISyntaxException e) {
42  				throw new RuntimeException(e);
43  			}
44  		}
45  
46  		// all the method does is:
47  		return new URIImpl(uriString);
48  	}
49  
50  	public static URI createURIWithoutChecking(String uriString) {
51  		assert java.net.URI.create(uriString) != null : "assert valid URI syntax";
52  		return new URIImpl(uriString);
53  	}

71  	public boolean equals(Object other) {
72  
73  		log
74  				.debug("comparing for equal: " + this + " and " + other + " of type "
75  						+ other.getClass());
76  
77  		if (other instanceof URI) {
78  			boolean equal = ((URI) other).toString().equals(this.toString());
79  
80  			log.debug("they are equal? " + equal);
81  
82  			return equal;
83  		} else
84  			log.debug("URIImpl cannot compare with type " + other.getClass() + " so this is false");
85  
86  		return false;
87  	}
