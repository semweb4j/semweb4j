20  /**
21   * RDF2Go adapters have toi implement this interface to be able to create Models and ModelSets.
22   * 
23   * The reasoning setting is set via a property named 'Reasoning'.
24   * @author voelkel
25   *
26   */
27  public interface ModelFactory
28  {
29  	/**
30  	 * The Property key to indicate he Reasoning state in the properties.

33  	
34

37  	// Model
38  	
39  	/**
40  	 * Create a default in-memory ModelSet with no inferencing.
41  	 * 
42  	 * @return a ModelSet implementation.
43  	 * @throws ModelException
44  	 *             if the adapter could not create the model
45  	 */
46      Model createModel() throws ModelException;
47      
48  	/**
49  	 * Create a new Model with inferencing. Type of reasoning is passed.
50  	 * 
51  	 * @param reasoning
52  	 *            the type of reasoning that is needed
53  	 * @return a model with reasoning support
54  	 * @throws ModelException
55  	 *             if the adapter could not create the Model
56  	 * @throws ReasoningNotSupportedException
57  	 *             if the passed kind of reasoning is not supported.
58  	 */
59      Model createModel(Reasoning reasoning) throws ReasoningNotSupportedException, ModelException;
60  
61  	/**
62  	 * Create a Model configured by the given properties. You have to look at
63  	 * the adapter documentation to see which properties do what.
64  	 * 
65  	 * @throws ModelException
66  	 *             if the adapter could not create the Model
67  	 * @throws ReasoningNotSupportedException
68  	 *             if the passed kind of reasoning is not supported.
69  	 * @param properties
70  	 *            configures the to-be-created Model
71  	 * @return the created Model
72  	 * @throws ModelException
73  	 */
74  	Model createModel(Properties p) throws ModelException;
75  
76  	/////////////////////////////

78  	
79      
80      /**
81       * create a default in-memory ModelSet with no inferencing.
82       * @return a ModelSet implementation.
83       * @throws NotImplementedException a RuntimeException if this method is not implemented. 
84       * Should not happen.

86      ModelSet createModelSet() throws ModelException;
87      
88  	/**

100 	 */
101     ModelSet createModelSet(Reasoning reasoning) throws ReasoningNotSupportedException, ModelException;
102     
103 	/**

106 	 * 
107 	 * @param properties -

109 	 * @return the created ModelSet
110 	 * @throws ModelException
111 	 *             if the adapter could not create the Model
112 	 * @throws ReasoningNotSupportedException
113 	 *             if the passed kind of reasoning is not supported.
114 	 */
115 	ModelSet createModelSet(Properties p) throws ModelException;
