19  /**
20   * A model where you can list all statements, find statements and check if a
21   * statement is contained.
22   * 
23   * @author voelkel
24   * 
25   */
26  public interface FindableModel {
27  
28  	/**
29  	 * get all statements in the model with this subject, predicate and object.
30  	 * Each of those (s,p,o) can be Variable.ANY
31  	 * 
32  	 * Iterator must be auto-close, i.e. when last element is fetched, the
33  	 * implementation must call close().
34  	 * 
35  	 * @param subject
36  	 *            URI or Object (= blankNode) or Variable
37  	 * @param predicate
38  	 *            URI or Variable
39  	 * @param object
40  	 *            URI or String (=plainLiteral) or BlankNode (=blankNode) or
41  	 *            TypedLiteral or LanguageTagLiteral or Variable
42  	 * @return a statement iterator
43  	 * @throws ModelException
44  	 */
45  	public ClosableIterable<? extends Statement> findStatements(ResourceOrVariable subject,
46  			UriOrVariable predicate, NodeOrVariable object) throws ModelException;
47  
48  	/**
49  	 * Iterator must be auto-close, i.e. when last element is fetched, the
50  	 * implementation must call close().
51  	 * @param pattern
52  	 * @return statement iterator returning all triples matching the given
53  	 *          pattern
54  	 * @throws ModelException
55  	 */
56  	public ClosableIterable<? extends Statement> findStatements(TriplePattern pattern) throws ModelException;
57  
58  	// ////////////////////////////
59  	// counting
60  
61  	public int countStatements(TriplePattern pattern) throws ModelException;
62  
63  	/**
64  	 * @param subject
65  	 * @param predicate
66  	 * @param object
67  	 * @return true if the statement (subject, predicate, object) is in the
68  	 *         model
69  	 * @throws ModelException
70  	 */
71  	public boolean contains(ResourceOrVariable subject, UriOrVariable predicate,
72  			NodeOrVariable object) throws ModelException;
73  
74  	public boolean contains(ResourceOrVariable subject, UriOrVariable predicate, String plainLiteral)
75  			throws ModelException;
76  
77  	public boolean contains(Statement s) throws ModelException;
78  
79  }
